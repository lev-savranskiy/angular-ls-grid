Local data, tableWidth, itemsPerPage, sort by start_date DESC, and 3 headers passed 

var app = angular.module('app', ['lsGrid']);
app.controller("demo", function ($scope) {
    var data = [{
        "id": 577,
        "name": "Realcube",
        "start_date": "2016-12-30",
        "clicks": 31
    }, {
        "id": 561,
        "name": "JumpXS",
        "start_date": "2017-02-15",
        "clicks": 85
    },
    var data = [{
        "id": 577,
        "name": "Realcube",
        "start_date": "2016-12-30",
        "clicks": 31
    }, {
        "id": 561,
        "name": "JumpXS",
        "start_date": "2017-02-15",
        "clicks": 85
    } ... ]

    $scope.options = {
        tableWidth: '600px',
        itemsPerPage: 7,
        sorter: {
           field:  'start_date',
           dir:  'desc'
        },
        headers: {"name": "Name", "start_date": "Start Date", "clicks": "Clicks"},
        data: data
    };
});
<ls-grid options="options"></ls-grid>
